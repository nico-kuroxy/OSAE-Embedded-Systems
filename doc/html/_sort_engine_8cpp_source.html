83  unsigned long l = 0;
84  unsigned short savedValue;
85 
86  if (len < 2) return;
87  l=(len >> l);

103  while (child < heapSize) {

105  child++;
106  if (savedValue < toSort[child]) {
107  toSort[parent]=toSort[child];
108  parent=child;
109  child=(child<<1)+1;

111  }
112  else{

114  }
115  }
